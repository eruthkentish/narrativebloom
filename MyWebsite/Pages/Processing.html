import React, { useState, useEffect } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { Textarea } from "@/components/ui/textarea";

import { BookOpen, Heart, Eye, Mountain } from "lucide-react";


const emotionalRealms = [

  {

    id: "grief",

    name: "Realm of Grief",

    color: "from-indigo-900 to-purple-900",

    icon: Heart,

    description: "A space to process loss and mourning"

  },

  {

    id: "shame",

    name: "Realm of Shame", 

    color: "from-red-900 to-orange-900",

    icon: Eye,

    description: "Confronting self-judgment and acceptance"

  },

  {

    id: "hope",

    name: "Realm of Hope",

    color: "from-blue-900 to-teal-900", 

    icon: Mountain,

    description: "Finding light in darkness"

  }

];


const prompts = {

  grief: [

    {

      id: 1,

      title: "The Memory",

      text: "What moment of loss still echoes in your heart? Write a single line that captures its essence."

    },

    {

      id: 2, 

      title: "The Bridge",

      text: "Grief builds walls, but also bridges. What word would you use to cross from pain to understanding?"

    }

  ],

  shame: [

    {

      id: 1,

      title: "The Shadow",

      text: "What story do you tell yourself that isn't true? Write it down to release its power."

    }

  ],

  hope: [

    {

      id: 1,

      title: "The Dawn",

      text: "In your darkest moment, what small light did you hold onto? Name it."

    }

  ]

};


export default function Processing() {

  const [gameState, setGameState] = useState("intro"); // intro, realm_select, playing, prompt

  const [currentRealm, setCurrentRealm] = useState(null);

  const [currentPrompt, setCurrentPrompt] = useState(null);

  const [userWords, setUserWords] = useState([]);

  const [promptText, setPromptText] = useState("");

  const [realmProgress, setRealmProgress] = useState({});


  const startGame = () => {

    setGameState("realm_select");

  };


  const enterRealm = (realm) => {

    setCurrentRealm(realm);

    setCurrentPrompt(prompts[realm.id][0]);

    setGameState("prompt");

  };


  const submitPrompt = () => {

    if (!promptText.trim()) return;


    const newWord = {

      realmId: currentRealm.id,

      promptId: currentPrompt.id,

      text: promptText,

      timestamp: Date.now()

    };


    setUserWords([...userWords, newWord]);

    

    // Update realm progress

    const realmWords = userWords.filter(w => w.realmId === currentRealm.id).length + 1;

    setRealmProgress({

      ...realmProgress,

      [currentRealm.id]: realmWords

    });


    setPromptText("");

    

    // Check if there are more prompts

    const nextPromptIndex = prompts[currentRealm.id].findIndex(p => p.id === currentPrompt.id) + 1;

    if (nextPromptIndex < prompts[currentRealm.id].length) {

      setCurrentPrompt(prompts[currentRealm.id][nextPromptIndex]);

    } else {

      setGameState("realm_complete");

    }

  };


  const returnToRealmSelect = () => {

    setCurrentRealm(null);

    setCurrentPrompt(null);

    setGameState("realm_select");

  };


  const IntroScreen = () => (

    <motion.div

      initial={{ opacity: 0 }}

      animate={{ opacity: 1 }}

      className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-800 flex items-center justify-center p-6"

    >

      <div className="text-center max-w-2xl">

        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          transition={{ delay: 0.2 }}

          className="mb-8"

        >

          <BookOpen className="w-16 h-16 text-purple-300 mx-auto mb-4" />

          <h1 className="text-6xl font-serif italic text-white mb-4">Processing</h1>

          <p className="text-xl text-purple-200">The Emotional Healing Behind Words</p>

        </motion.div>

        

        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          transition={{ delay: 0.5 }}

          className="mb-8"

        >

          <p className="text-lg text-gray-300 leading-relaxed mb-6">

            An introspective journey through your emotional landscapes. 

            Move through realms of feeling and use your words to transform pain into power, 

            introspection into liberation.

          </p>

          <p className="text-gray-400">

            This is not a game about winning. It's about feeling, expressing, and healing.

          </p>

        </motion.div>


        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          transition={{ delay: 0.8 }}

        >

          <Button 

            onClick={startGame}

            className="bg-purple-600 hover:bg-purple-500 text-white px-8 py-3 text-lg"

          >

            Begin Your Journey

          </Button>

        </motion.div>

      </div>

    </motion.div>

  );


  const RealmSelectScreen = () => (

    <motion.div

      initial={{ opacity: 0 }}

      animate={{ opacity: 1 }}

      className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-800 p-6"

    >

      <div className="max-w-4xl mx-auto">

        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          className="text-center mb-12"

        >

          <h2 className="text-4xl font-serif text-white mb-4">Choose Your Realm</h2>

          <p className="text-gray-300">Each realm represents a different emotional landscape to explore</p>

        </motion.div>


        <div className="grid md:grid-cols-3 gap-8">

          {emotionalRealms.map((realm, index) => (

            <motion.div

              key={realm.id}

              initial={{ y: 50, opacity: 0 }}

              animate={{ y: 0, opacity: 1 }}

              transition={{ delay: index * 0.2 }}

              className={`bg-gradient-to-br ${realm.color} rounded-2xl p-8 border border-white/10 hover:border-white/30 transition-all duration-300 cursor-pointer group`}

              onClick={() => enterRealm(realm)}

            >

              <realm.icon className="w-12 h-12 text-white mb-4 group-hover:scale-110 transition-transform" />

              <h3 className="text-xl font-semibold text-white mb-3">{realm.name}</h3>

              <p className="text-gray-200 mb-4">{realm.description}</p>

              

              {realmProgress[realm.id] && (

                <div className="text-sm text-white/70">

                  Progress: {realmProgress[realm.id]} words written

                </div>

              )}

              

              <div className="mt-4 text-white/60 text-sm">

                Click to enter →

              </div>

            </motion.div>

          ))}

        </div>


        {userWords.length > 0 && (

          <motion.div

            initial={{ y: 30, opacity: 0 }}

            animate={{ y: 0, opacity: 1 }}

            className="mt-12 bg-white/5 rounded-2xl p-6"

          >

            <h3 className="text-xl font-semibold text-white mb-4">Your Journey So Far</h3>

            <div className="grid gap-4">

              {userWords.map((word, index) => (

                <div key={index} className="bg-white/10 rounded-lg p-4">

                  <div className="text-sm text-gray-300 mb-2">

                    {emotionalRealms.find(r => r.id === word.realmId)?.name}

                  </div>

                  <div className="text-white italic">"{word.text}"</div>

                </div>

              ))}

            </div>

          </motion.div>

        )}

      </div>

    </motion.div>

  );


  const PromptScreen = () => (

    <motion.div

      initial={{ opacity: 0 }}

      animate={{ opacity: 1 }}

      className={`min-h-screen bg-gradient-to-br ${currentRealm.color} flex items-center justify-center p-6`}

    >

      <div className="max-w-2xl w-full">

        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          className="bg-black/30 backdrop-blur-sm rounded-2xl p-8 border border-white/20"

        >

          <div className="text-center mb-8">

            <currentRealm.icon className="w-12 h-12 text-white mx-auto mb-4" />

            <h2 className="text-3xl font-serif text-white mb-2">{currentPrompt.title}</h2>

            <p className="text-gray-200">{currentRealm.name}</p>

          </div>


          <div className="mb-8">

            <p className="text-lg text-gray-100 leading-relaxed mb-6">

              {currentPrompt.text}

            </p>

            

            <Textarea

              value={promptText}

              onChange={(e) => setPromptText(e.target.value)}

              placeholder="Write your truth here..."

              className="bg-white/10 border-white/30 text-white placeholder-gray-300 focus:border-white/50 focus:ring-white/30 min-h-[120px]"

              autoFocus

            />

          </div>


          <div className="flex justify-between">

            <Button

              onClick={returnToRealmSelect}

              variant="outline"

              className="bg-transparent border-white/30 text-white hover:bg-white/10"

            >

              Return to Realms

            </Button>

            <Button

              onClick={submitPrompt}

              disabled={!promptText.trim()}

              className="bg-white/20 hover:bg-white/30 text-white disabled:opacity-50"

            >

              Continue Journey

            </Button>

          </div>

        </motion.div>

      </div>

    </motion.div>

  );


  const RealmCompleteScreen = () => (

    <motion.div

      initial={{ opacity: 0 }}

      animate={{ opacity: 1 }}

      className={`min-h-screen bg-gradient-to-br ${currentRealm.color} flex items-center justify-center p-6`}

    >

      <div className="max-w-2xl w-full text-center">

        <motion.div

          initial={{ y: 30, opacity: 0 }}

          animate={{ y: 0, opacity: 1 }}

          className="bg-black/30 backdrop-blur-sm rounded-2xl p-8 border border-white/20"

        >

          <currentRealm.icon className="w-16 h-16 text-white mx-auto mb-6" />

          <h2 className="text-4xl font-serif text-white mb-4">Realm Complete</h2>

          <p className="text-xl text-gray-200 mb-8">

            You have journeyed through the {currentRealm.name}. 

            Your words have become bridges across the emotional landscape.

          </p>

          

          <div className="bg-white/10 rounded-lg p-6 mb-8">

            <h3 className="text-lg font-semibold text-white mb-4">Your Words in This Realm:</h3>

            {userWords

              .filter(w => w.realmId === currentRealm.id)

              .map((word, index) => (

                <div key={index} className="text-white italic mb-2">

                  "{word.text}"

                </div>

              ))}

          </div>


          <Button

            onClick={returnToRealmSelect}

            className="bg-white/20 hover:bg-white/30 text-white"

          >

            Explore Another Realm

          </Button>

        </motion.div>

      </div>

    </motion.div>

  );


  return (

    <div className="relative">

      <AnimatePresence mode="wait">

        {gameState === "intro" && <IntroScreen />}

        {gameState === "realm_select" && <RealmSelectScreen />}

        {gameState === "prompt" && <PromptScreen />}

        {gameState === "realm_complete" && <RealmCompleteScreen />}

      </AnimatePresence>

    </div>

  );

}
