import React, { useState, useEffect, useCallback } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Link, useLocation } from "react-router-dom"; // Added useLocation

import { createPageUrl } from "@/utils";

import { User } from "@/entities/User";

import { WritingEntry } from "@/entities/WritingEntry";

import { Button } from "@/components/ui/button";

import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";

import { PenSquare, ArrowRight, BookOpen, BrainCircuit, HeartPulse, Sparkles, Loader2, Book } from "lucide-react";

import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";

import Journal from "../components/dashboard/Journal";


export default function Dashboard() {

  const [user, setUser] = useState(null);

  const [entries, setEntries] = useState([]);

  const [isLoading, setIsLoading] = useState(true);

  const location = useLocation(); // Initialize useLocation hook

  const [activeTab, setActiveTab] = useState('overview'); // State to manage active tab


  const loadDashboardData = useCallback(async () => {

    try {

      const currentUser = await User.me();

      setUser(currentUser);

      const userEntries = await WritingEntry.filter({ created_by: currentUser.email }, "-created_date");

      setEntries(userEntries);

    } catch (error) {

      console.error("Error loading dashboard data:", error);

    } finally {

      setIsLoading(false);

    }

  }, []);


  useEffect(() => {

    loadDashboardData();

  }, [loadDashboardData]);


  // Effect to read URL query parameters and set the active tab

  useEffect(() => {

    const params = new URLSearchParams(location.search);

    const tab = params.get('tab');

    if (tab === 'journal') {

      setActiveTab('journal');

    } else {

      setActiveTab('overview');

    }

  }, [location.search]); // Re-run when location.search changes


  if (isLoading) {

    return (

      <div className="flex justify-center items-center min-h-screen bg-slate-50">

        <Loader2 className="h-10 w-10 animate-spin text-slate-400" />

        <p className="ml-3 text-slate-600">Loading your dashboard...</p>

      </div>

    );

  }


  const featureCards = [

    {

      title: "Writing Studio",

      description: "Begin your journey with guided prompts or free writing.",

      icon: PenSquare,

      path: "WritingStudio",

      color: "from-blue-500 to-indigo-600"

    },

    {

      title: "Poetry Garden",

      description: "Visualize your emotional growth in a living garden.",

      icon: HeartPulse,

      path: "PoetryGarden",

      color: "from-emerald-500 to-teal-600"

    },

    {

      title: "Reflection",

      description: "Analyze your writing with our 3-process framework.",

      icon: BrainCircuit,

      path: "Reflection",

      color: "from-purple-500 to-violet-600"

    }

  ];


  return (

    <div className="min-h-screen bg-slate-50">

      <div className="max-w-7xl mx-auto p-6">

        {/* Header */}

        <motion.div

          initial={{ opacity: 0, y: 20 }}

          animate={{ opacity: 1, y: 0 }}

          className="mb-8"

        >

          <h1 className="text-3xl font-bold text-slate-800 mb-2">Welcome back, {user?.full_name || 'Writer'}</h1>

          <p className="text-slate-600">Continue your healing journey through words</p>

        </motion.div>


        {/* Main Dashboard Tabs */}

        <Tabs value={activeTab} onValueChange={setActiveTab} className="w-full"> {/* Controlled Tabs component */}

          <TabsList className="grid w-full grid-cols-2 bg-white/60 p-2 rounded-2xl mb-8">

            <TabsTrigger value="overview" className="rounded-xl">Dashboard Overview</TabsTrigger>

            <TabsTrigger value="journal" className="rounded-xl">Journal & Writings</TabsTrigger>

          </TabsList>


          {/* Dashboard Overview */}

          <TabsContent value="overview">

            <div className="grid md:grid-cols-3 gap-6">

              {featureCards.map((card, index) => (

                <motion.div

                  key={index}

                  initial={{ opacity: 0, y: 20 }}

                  animate={{ opacity: 1, y: 0 }}

                  transition={{ delay: 0.1 + index * 0.1 }}

                >

                  <Card className="card-therapy h-full hover:shadow-lg transition-all duration-300">

                    <CardHeader>

                      <div className={`w-12 h-12 bg-gradient-to-br ${card.color} rounded-xl flex items-center justify-center mb-4`}>

                        <card.icon className="w-6 h-6 text-white" />

                      </div>

                      <CardTitle className="text-xl text-slate-800">{card.title}</CardTitle>

                      <CardDescription className="text-slate-600">{card.description}</CardDescription>

                    </CardHeader>

                    <CardContent>

                      <Link to={createPageUrl(card.path)}>

                        <Button className="w-full btn-therapy">

                          Explore <ArrowRight className="ml-2 w-4 h-4" />

                        </Button>

                      </Link>

                    </CardContent>

                  </Card>

                </motion.div>

              ))}

            </div>


            {/* Quick Stats */}

            <motion.div

              initial={{ opacity: 0, y: 20 }}

              animate={{ opacity: 1, y: 0 }}

              transition={{ delay: 0.4 }}

              className="mt-8"

            >

              <Card className="card-therapy">

                <CardHeader>

                  <CardTitle className="flex items-center text-xl text-slate-800">

                    <Book className="w-5 h-5 mr-3" />

                    Writing Progress

                  </CardTitle>

                </CardHeader>

                <CardContent>

                  <div className="grid grid-cols-3 gap-6 text-center">

                    <div>

                      <div className="text-2xl font-bold text-blue-600">{entries.length}</div>

                      <div className="text-sm text-slate-600">Total Entries</div>

                    </div>

                    <div>

                      <div className="text-2xl font-bold text-emerald-600">

                        {entries.reduce((sum, entry) => sum + (entry.word_count || 0), 0).toLocaleString()}

                      </div>

                      <div className="text-sm text-slate-600">Words Written</div>

                    </div>

                    <div>

                      <div className="text-2xl font-bold text-purple-600">7</div>

                      <div className="text-sm text-slate-600">Day Streak</div>

                    </div>

                  </div>

                </CardContent>

              </Card>

            </motion.div>

          </TabsContent>


          {/* Journal & Writings */}

          <TabsContent value="journal">

            <Journal entries={entries} onRefresh={loadDashboardData} />

          </TabsContent>

        </Tabs>

      </div>

    </div>

  );

}
