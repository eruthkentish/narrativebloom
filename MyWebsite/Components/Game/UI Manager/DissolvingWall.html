import React, { useState } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { Input } from "@/components/ui/input";

import { Card, CardContent } from "@/components/ui/card";

import { Shield, Zap, Sparkles } from "lucide-react";


export default function DissolvingWall({ 

  position, 

  triggerWords, 

  prompt, 

  onDissolve 

}) {

  const [currentWord, setCurrentWord] = useState("");

  const [isActive, setIsActive] = useState(false);

  const [isDissolved, setIsDissolved] = useState(false);

  const [attempts, setAttempts] = useState([]);

  const [wallIntegrity, setWallIntegrity] = useState(100);


  const handleWordSubmit = () => {

    const word = currentWord.toLowerCase().trim();

    

    if (triggerWords.includes(word)) {

      setIsDissolved(true);

      setWallIntegrity(0);

      setIsActive(false);

      onDissolve({

        type: 'wall',

        triggerWord: word,

        attempts: attempts.length + 1

      });

    } else {

      const damage = Math.random() * 10 + 5; // 5-15% damage for wrong words

      setWallIntegrity(prev => Math.max(0, prev - damage));

      setAttempts([...attempts, word]);

    }

    

    setCurrentWord("");

  };


  const handleKeyPress = (e) => {

    if (e.key === 'Enter') {

      handleWordSubmit();

    }

  };


  const getWallOpacity = () => {

    return isDissolved ? 0 : wallIntegrity / 100;

  };


  const getCrackPattern = () => {

    const damage = 100 - wallIntegrity;

    if (damage < 20) return "";

    if (damage < 50) return "opacity-20";

    if (damage < 80) return "opacity-40";

    return "opacity-60";

  };


  return (

    <>

      {/* Wall in the world */}

      <motion.div

        className="absolute cursor-pointer z-30"

        style={{

          left: `${position.x}%`,

          top: `${position.y}%`,

          transform: 'translate(-50%, -50%)'

        }}

        animate={{

          opacity: getWallOpacity(),

          scale: isDissolved ? [1, 1.1, 0] : 1

        }}

        transition={{

          opacity: { duration: 2 },

          scale: { duration: 1.5, times: [0, 0.3, 1] }

        }}

        onClick={() => !isDissolved && setIsActive(true)}

      >

        {/* Wall structure */}

        <div className="relative w-20 h-32 flex items-center justify-center">

          {/* Main wall */}

          <div className="absolute inset-0 bg-gradient-to-b from-stone-700 via-stone-600 to-stone-800 rounded-lg shadow-xl" />

          

          {/* Stone blocks */}

          <div className="absolute inset-2 grid grid-cols-3 gap-1">

            {Array.from({ length: 9 }).map((_, i) => (

              <motion.div

                key={i}

                className="bg-stone-500/70 rounded-sm"

                animate={{

                  opacity: wallIntegrity > (i * 10 + 10) ? 1 : 0.3,

                  scale: wallIntegrity > (i * 10 + 10) ? 1 : 0.8

                }}

              />

            ))}

          </div>

          

          {/* Cracks overlay */}

          {wallIntegrity < 100 && (

            <div className={`absolute inset-0 ${getCrackPattern()}`}>

              <svg viewBox="0 0 80 128" className="w-full h-full">

                <path

                  d="M10 20 L25 35 L40 25 L60 40 L70 30 M15 50 L35 65 L55 55 L70 70 M20 80 L40 95 L65 85 M25 100 L50 115"

                  stroke="rgba(0,0,0,0.6)"

                  strokeWidth="1"

                  fill="none"

                />

              </svg>

            </div>

          )}

          

          {/* Wall icon */}

          <div className="absolute inset-0 flex items-center justify-center">

            <Shield className="w-10 h-10 text-stone-300 opacity-80" />

          </div>

        </div>

        

        {/* Interaction hint */}

        {!isDissolved && (

          <div className="absolute -bottom-8 left-1/2 transform -translate-x-1/2 bg-black/60 text-white text-xs px-2 py-1 rounded-md whitespace-nowrap">

            Speak to dissolve

          </div>

        )}

      </motion.div>


      {/* Word Input Interface */}

      <AnimatePresence>

        {isActive && !isDissolved && (

          <motion.div

            initial={{ opacity: 0 }}

            animate={{ opacity: 1 }}

            exit={{ opacity: 0 }}

            className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center p-6 z-50"

          >

            <motion.div

              initial={{ scale: 0.9, y: 50 }}

              animate={{ scale: 1, y: 0 }}

              exit={{ scale: 0.9, y: 50 }}

              className="w-full max-w-md"

            >

              <Card className="bg-gradient-to-br from-slate-900/95 to-stone-900/95 border-stone-600/50 text-white">

                <CardContent className="p-6 space-y-6">

                  <div className="text-center">
