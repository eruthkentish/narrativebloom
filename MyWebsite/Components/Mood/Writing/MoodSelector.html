import React from 'react';

import { motion } from 'framer-motion';


const moodEmojis = [

  { value: 1, emoji: "😞", label: "Struggling" },

  { value: 2, emoji: "😐", label: "Low" },

  { value: 3, emoji: "🙂", label: "Okay" },

  { value: 4, emoji: "😊", label: "Good" },

  { value: 5, emoji: "😄", label: "Great" }

];


export default function MoodSelector({ title, onMoodSelect, selectedMood }) {

  return (

    <div className="w-full max-w-lg mx-auto text-center">

      <h3 className="text-xl font-semibold text-gray-700 mb-6">{title}</h3>

      <div className="flex justify-around items-center bg-white/50 p-4 rounded-2xl shadow-inner-soft">

        {moodEmojis.map((item) => (

          <motion.button

            key={item.value}

            onClick={() => onMoodSelect(item.value)}

            className="flex flex-col items-center p-2 rounded-xl transition-all"

            whileHover={{ scale: 1.1 }}

            animate={selectedMood === item.value ? { scale: 1.15, y: -5 } : { scale: 1, y: 0 }}

          >

            <span className="text-4xl mb-1 transition-transform duration-300">{item.emoji}</span>

            <span className={`text-xs font-medium transition-colors ${selectedMood === item.value ? 'text-rose-600' : 'text-gray-500'}`}>

              {item.label}

            </span>

            {selectedMood === item.value && (

              <motion.div

                layoutId="underline"

                className="h-1 w-6 bg-rose-500 rounded-full mt-2"

              />

            )}

          </motion.button>

        ))}

      </div>

    </div>

  );

}
