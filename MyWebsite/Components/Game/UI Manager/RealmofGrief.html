import React, { useState, useEffect } from "react";

import { motion, AnimatePresence } from "framer-motion";

import MemoryArtifact from "../MemoryArtifact";

import WordBridge from "../WordBridge";

import DissolvingWall from "../DissolvingWall";


export default function RealmOfGrief() {

  const [timeOfDay, setTimeOfDay] = useState(0); // 0-1, controls lighting

  const [rainIntensity, setRainIntensity] = useState(0.3);

  const [playerDiscoveries, setPlayerDiscoveries] = useState([]);


  useEffect(() => {

    // Gentle day/night cycle

    const interval = setInterval(() => {

      setTimeOfDay(prev => (prev + 0.001) % 1);

    }, 100);

    return () => clearInterval(interval);

  }, []);


  const handleDiscovery = (discovery) => {

    setPlayerDiscoveries(prev => [...prev, discovery]);

    // Reduce rain intensity as player processes grief

    setRainIntensity(prev => Math.max(0, prev - 0.1));

  };


  const getLighting = () => {

    const darkness = Math.sin(timeOfDay * Math.PI * 2) * 0.3 + 0.7;

    return {

      filter: `brightness(${darkness}) contrast(1.1) saturate(0.8)`,

      background: `linear-gradient(180deg, 

        rgba(15, 23, 42, ${0.9 - timeOfDay * 0.3}) 0%, 

        rgba(30, 41, 59, ${0.8 - timeOfDay * 0.2}) 50%, 

        rgba(51, 65, 85, ${0.7 - timeOfDay * 0.1}) 100%)`

    };

  };


  return (

    <div className="relative min-h-screen overflow-hidden" style={getLighting()}>

      {/* Background Landscape */}

      <div className="absolute inset-0">

        {/* Misty Mountains */}

        <div className="absolute inset-0 bg-gradient-to-b from-slate-600 via-slate-700 to-slate-800">

          <motion.div

            animate={{

              backgroundPosition: ['0% 0%', '100% 100%']

            }}

            transition={{

              duration: 60,

              repeat: Infinity,

              ease: "linear"

            }}

            className="absolute inset-0 opacity-20"

            style={{

              backgroundImage: `url("data:image/svg+xml,%3Csvg width='40' height='40' viewBox='0 0 40 40' xmlns='http://www.w3.org/2000/svg'%3E%3Cg fill='%23ffffff' fill-opacity='0.1'%3E%3Cpath d='M20 0l20 20-20 20L0 20z'/%3E%3C/g%3E%3C/svg%3E")`,

              backgroundSize: '40px 40px'

            }}

          />

        </div>


        {/* Flowing River */}

        <motion.div

          animate={{

            backgroundPosition: ['0% 0%', '200% 0%']

          }}

          transition={{

            duration: 8,

            repeat: Infinity,

            ease: "linear"

          }}

          className="absolute bottom-0 left-0 right-0 h-32 bg-gradient-to-r from-blue-800/60 to-slate-600/60"

          style={{

            backgroundImage: `linear-gradient(90deg, transparent 0%, rgba(59, 130, 246, 0.3) 50%, transparent 100%)`,

            backgroundSize: '200px 100%'

          }}

        />


        {/* Gentle Rain */}

        {rainIntensity > 0 && (

          <div className="absolute inset-0 pointer-events-none">

            {Array.from({ length: Math.floor(rainIntensity * 50) }).map((_, i) => (

              <motion.div

                key={i}

                initial={{ 

                  y: -10, 

                  x: Math.random() * window.innerWidth,

                  opacity: 0 

                }}

                animate={{ 

                  y: window.innerHeight + 10,

                  opacity: [0, 0.6, 0] 

                }}

                transition={{ 

                  duration: 3 + Math.random() * 2,

                  repeat: Infinity,

                  delay: Math.random() * 5

                }}

                className="absolute w-px h-6 bg-gradient-to-b from-transparent via-blue-200/50 to-transparent"

              />

            ))}

          </div>

        )}


        {/* Floating Memory Fragments */}

        <div className="absolute inset-0">

          {Array.from({ length: 5 }).map((_, i) => (

            <motion.div

              key={i}

              animate={{

                y: [-20, 20, -20],

                x: [-10, 10, -10],

                rotate: [0, 5, -5, 0]

              }}

              transition={{

                duration: 8 + i * 2,

                repeat: Infinity,

                ease: "easeInOut",

                delay: i * 1.5

              }}

              className="absolute w-8 h-8 bg-white/10 rounded-lg backdrop-blur-sm"

              style={{

                left: `${20 + i * 15}%`,

                top: `${30 + i * 10}%`

              }}

            />

          ))}

        </div>

      </div>


      {/* Interactive Elements */}

      <MemoryArtifact

        position={{ x: 25, y: 60 }}

        type="letter"

        emotion="grief"

        prompt="Write a letter to someone you've lost. What would you want them to know?"

        onInteract={handleDiscovery}

      />


      <MemoryArtifact

        position={{ x: 70, y: 40 }}

        type="photo"

        emotion="grief"

        prompt="Describe a moment of joy you shared with them. Let that light shine through the sorrow."

        onInteract={handleDiscovery}

      />
