import React, { useState } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { 

  X, 

  Volume2, 

  VolumeX, 

  Book, 

  Heart, 

  Feather,

  Pause,

  Play,

  Home

} from "lucide-react";


export default function UIManager({ 

  playerProgress, 

  currentRealm, 

  onExit, 

  soundEnabled, 

  setSoundEnabled 

}) {

  const [isPaused, setIsPaused] = useState(false);

  const [showProgress, setShowProgress] = useState(false);


  const realmNames = {

    grief: 'The Valley of Letting Go',

    shame: 'The Shadow Gardens',

    abandonment: 'The Echoing Chambers',

    forgiveness: 'The Healing Springs',

    hope: 'The Dawn Meadows'

  };


  return (

    <>

      {/* Top UI Bar */}

      <div className="fixed top-0 left-0 right-0 z-50 bg-gradient-to-b from-black/60 to-transparent p-6">

        <div className="flex justify-between items-start">

          <div className="bg-black/40 backdrop-blur-sm rounded-2xl p-4 text-white">

            <div className="flex items-center space-x-3 mb-2">

              <div className="w-3 h-3 bg-gradient-to-r from-purple-400 to-pink-500 rounded-full animate-pulse" />

              <h3 className="font-serif text-lg">

                {realmNames[currentRealm]}

              </h3>

            </div>

            <div className="text-sm opacity-75">

              Words written: {playerProgress.writtenWords}

            </div>

          </div>


          <div className="flex items-center space-x-3">

            <button

              onClick={() => setShowProgress(!showProgress)}

              className="p-3 bg-black/40 backdrop-blur-sm rounded-xl text-white hover:bg-black/60 transition-colors"

            >

              <Book className="w-5 h-5" />

            </button>


            <button

              onClick={() => setSoundEnabled(!soundEnabled)}

              className="p-3 bg-black/40 backdrop-blur-sm rounded-xl text-white hover:bg-black/60 transition-colors"

            >

              {soundEnabled ? <Volume2 className="w-5 h-5" /> : <VolumeX className="w-5 h-5" />}

            </button>


            <button

              onClick={() => setIsPaused(!isPaused)}

              className="p-3 bg-black/40 backdrop-blur-sm rounded-xl text-white hover:bg-black/60 transition-colors"

            >

              {isPaused ? <Play className="w-5 h-5" /> : <Pause className="w-5 h-5" />}

            </button>


            <button

              onClick={onExit}

              className="p-3 bg-black/40 backdrop-blur-sm rounded-xl text-white hover:bg-red-600/60 transition-colors"

            >

              <Home className="w-5 h-5" />

            </button>

          </div>

        </div>

      </div>


      {/* Progress Panel */}

      <AnimatePresence>

        {showProgress && (

          <motion.div

            initial={{ opacity: 0, x: 300 }}

            animate={{ opacity: 1, x: 0 }}

            exit={{ opacity: 0, x: 300 }}

            className="fixed right-6 top-24 bottom-6 w-80 bg-black/60 backdrop-blur-lg rounded-2xl p-6 text-white z-40 overflow-y-auto"

          >

            <div className="flex justify-between items-center mb-6">

              <h3 className="font-serif text-xl">Your Journey</h3>

              <button

                onClick={() => setShowProgress(false)}

                className="p-2 hover:bg-white/10 rounded-lg transition-colors"

              >

                <X className="w-4 h-4" />

              </button>

            </div>


            <div className="space-y-6">

              <div>

                <h4 className="font-semibold mb-3 flex items-center">

                  <Heart className="w-4 h-4 mr-2 text-pink-400" />

                  Emotional Progress

                </h4>

                <div className="space-y-2">

                  {playerProgress.emotionalJourney.slice(-5).map((entry, index) => (

                    <div key={index} className="bg-white/10 rounded-lg p-3">

                      <div className="text-sm font-medium">{entry.word}</div>

                      <div className="text-xs opacity-75">

                        {entry.emotion} • {realmNames[entry.realm]}

                      </div>

                    </div>

                  ))}

                </div>

              </div>


              <div>

                <h4 className="font-semibold mb-3 flex items-center">

                  <Feather className="w-4 h-4 mr-2 text-purple-400" />

                  Artifacts Collected

                </h4>

                <div className="text-sm opacity-75">

                  {playerProgress.artifacts.length} memories preserved

                </div>

              </div>


              <div>

                <h4 className="font-semibold mb-3">Realms Unlocked</h4>

                <div className="space-y-2">

                  {playerProgress.realmsUnlocked.map(realmId => (

                    <div key={realmId} className="flex items-center text-sm">

                      <div className="w-2 h-2 bg-green-400 rounded-full mr-3" />

                      {realmNames[realmId]}

                    </div>

                  ))}

                </div>

              </div>

            </div>

          </motion.div>

        )}

      </AnimatePresence>
