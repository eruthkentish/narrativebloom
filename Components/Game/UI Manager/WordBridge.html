import React, { useState } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { Input } from "@/components/ui/input";

import { Card, CardContent } from "@/components/ui/card";

import { Milestone, Check, X, Sparkles } from "lucide-react";


export default function WordBridge({ 

  position, 

  requiredWords, 

  prompt, 

  onComplete 

}) {

  const [currentWord, setCurrentWord] = useState("");

  const [foundWords, setFoundWords] = useState([]);

  const [isActive, setIsActive] = useState(false);

  const [isComplete, setIsComplete] = useState(false);

  const [attempts, setAttempts] = useState([]);


  const handleWordSubmit = () => {

    const word = currentWord.toLowerCase().trim();

    

    if (word && requiredWords.includes(word) && !foundWords.includes(word)) {

      const newFoundWords = [...foundWords, word];

      setFoundWords(newFoundWords);

      setAttempts([...attempts, { word, correct: true }]);

      

      if (newFoundWords.length === requiredWords.length) {

        setIsComplete(true);

        onComplete({

          type: 'bridge',

          words: newFoundWords,

          attempts: attempts.length + 1

        });

      }

    } else {

      setAttempts([...attempts, { word, correct: false }]);

    }

    

    setCurrentWord("");

  };


  const handleKeyPress = (e) => {

    if (e.key === 'Enter') {

      handleWordSubmit();

    }

  };


  const bridgeProgress = foundWords.length / requiredWords.length;


  return (

    <>

      {/* Bridge in the world */}

      <motion.div

        className="absolute cursor-pointer z-30"

        style={{

          left: `${position.x}%`,

          top: `${position.y}%`,

          transform: 'translate(-50%, -50%)'

        }}

        onClick={() => setIsActive(true)}

      >

        {/* Bridge Structure */}

        <div className="relative w-32 h-8">

          {/* Bridge base */}

          <div className="absolute inset-0 bg-gradient-to-r from-stone-600 via-stone-500 to-stone-600 rounded-lg shadow-lg" />

          

          {/* Progress overlay */}

          <motion.div

            className="absolute inset-0 bg-gradient-to-r from-emerald-400 to-teal-500 rounded-lg origin-left"

            style={{ scaleX: bridgeProgress }}

            transition={{ duration: 0.8, ease: "easeOut" }}

          />

          

          {/* Bridge icon */}

          <div className="absolute inset-0 flex items-center justify-center">

            <Milestone className={`w-6 h-6 ${isComplete ? 'text-white' : 'text-stone-200'}`} />

          </div>

          

          {/* Planks */}

          {Array.from({ length: requiredWords.length }).map((_, i) => (

            <motion.div

              key={i}

              className={`absolute w-6 h-1 rounded-sm ${

                i < foundWords.length 

                  ? 'bg-emerald-300/80' 

                  : 'bg-stone-700/50'

              }`}

              style={{

                left: `${(i / (requiredWords.length - 1)) * 80 + 10}%`,

                bottom: '2px'

              }}

              animate={{

                opacity: i < foundWords.length ? 1 : 0.5,

                y: i < foundWords.length ? 0 : 2

              }}

            />

          ))}

        </div>

        

        {/* Interaction hint */}

        {!isComplete && (

          <div className="absolute -bottom-8 left-1/2 transform -translate-x-1/2 bg-black/60 text-white text-xs px-2 py-1 rounded-md whitespace-nowrap">

            Build with words

          </div>

        )}

      </motion.div>


      {/* Word Building Interface */}

      <AnimatePresence>

        {isActive && !isComplete && (

          <motion.div

            initial={{ opacity: 0 }}

            animate={{ opacity: 1 }}

            exit={{ opacity: 0 }}

            className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center p-6 z-50"

          >

            <motion.div

              initial={{ scale: 0.9, y: 50 }}

              animate={{ scale: 1, y: 0 }}

              exit={{ scale: 0.9, y: 50 }}

              className="w-full max-w-md"

            >

              <Card className="bg-gradient-to-br from-slate-900/95 to-stone-900/95 border-stone-600/50 text-white">

                <CardContent className="p-6 space-y-6">

                  <div className="text-center">

                    <div className="w-12 h-12 bg-gradient-to-br from-stone-500 to-stone-700 rounded-xl flex items-center justify-center mx-auto mb-4">

                      <Milestone className="w-6 h-6 text-white" />

                    </div>

                    <h3 className="text-xl font-serif mb-2">Word Bridge</h3>

                    <p className="text-stone-300 text-sm italic">

                      {prompt}

                    </p>

                  </div>


                  {/* Progress */}

                  <div className="bg-stone-800/50 rounded-lg p-4">

                    <div className="flex justify-between items-center mb-2">

                      <span className="text-sm text-stone-300">Progress</span>

                      <span className="text-sm text-stone-300">

                        {foundWords.length} / {requiredWords.length}

                      </span>

                    </div>
