import React, { useState, useEffect } from "react";

import { motion } from "framer-motion";

import { Heart, Feather, Sparkles } from "lucide-react";


export default function Player({ realm, progress, setProgress }) {

  const [position, setPosition] = useState({ x: 50, y: 50 });

  const [energy, setEnergy] = useState(100);

  const [emotionalState, setEmotionalState] = useState('neutral');

  const [recentWords, setRecentWords] = useState([]);


  const movePlayer = (direction) => {

    setPosition(prev => {

      const speed = 5;

      let newPos = { ...prev };

      

      switch(direction) {

        case 'up': newPos.y = Math.max(10, prev.y - speed); break;

        case 'down': newPos.y = Math.min(90, prev.y + speed); break;

        case 'left': newPos.x = Math.max(10, prev.x - speed); break;

        case 'right': newPos.x = Math.min(90, prev.x + speed); break;

      }

      

      return newPos;

    });

  };


  const addWord = (word, emotion) => {

    setRecentWords(prev => [...prev.slice(-4), { word, emotion, id: Date.now() }]);

    setProgress(prevProgress => ({

      ...prevProgress,

      writtenWords: prevProgress.writtenWords + 1,

      emotionalJourney: [...prevProgress.emotionalJourney, { word, emotion, realm, timestamp: Date.now() }]

    }));

  };


  useEffect(() => {

    const handleKeyPress = (e) => {

      switch(e.key) {

        case 'ArrowUp':

        case 'w':

        case 'W':

          e.preventDefault();

          movePlayer('up');

          break;

        case 'ArrowDown':

        case 's':

        case 'S':

          e.preventDefault();

          movePlayer('down');

          break;

        case 'ArrowLeft':

        case 'a':

        case 'A':

          e.preventDefault();

          movePlayer('left');

          break;

        case 'ArrowRight':

        case 'd':

        case 'D':

          e.preventDefault();

          movePlayer('right');

          break;

      }

    };


    window.addEventListener('keydown', handleKeyPress);

    return () => window.removeEventListener('keydown', handleKeyPress);

  }, []);


  const getPlayerIcon = () => {

    switch(emotionalState) {

      case 'grieving': return '💧';

      case 'healing': return '🌱';

      case 'hopeful': return '✨';

      case 'peaceful': return '🕊️';

      default: return '👤';

    }

  };


  return (

    <>

      {/* Player Avatar */}

      <motion.div

        className="fixed z-50 pointer-events-none"

        style={{

          left: `${position.x}%`,

          top: `${position.y}%`,

          transform: 'translate(-50%, -50%)'

        }}

        animate={{

          scale: [1, 1.1, 1],

        }}

        transition={{

          duration: 2,

          repeat: Infinity,

          ease: "easeInOut"

        }}

      >

        <div className="relative">

          <div className="w-12 h-12 bg-gradient-to-br from-white/80 to-white/60 rounded-full flex items-center justify-center shadow-lg border-2 border-white/40 backdrop-blur-sm">

            <span className="text-2xl">{getPlayerIcon()}</span>

          </div>

          

          {/* Energy indicator */}

          <div className="absolute -top-2 -right-2 w-4 h-4 bg-gradient-to-br from-blue-400 to-blue-600 rounded-full flex items-center justify-center">

            <Heart className="w-2 h-2 text-white" />

          </div>

          

          {/* Recent words trail */}

          {recentWords.map((wordData, index) => (

            <motion.div

              key={wordData.id}

              initial={{ opacity: 0, scale: 0, y: 0 }}

              animate={{ 

                opacity: [0, 1, 0],

                scale: [0, 1, 0.8],

                y: [0, -20 - (index * 10), -40 - (index * 15)]

              }}

              transition={{ 

                duration: 3,

                times: [0, 0.3, 1]

              }}

              className="absolute top-0 left-1/2 transform -translate-x-1/2 pointer-events-none"

            >

              <div className="bg-gradient-to-r from-purple-500/80 to-pink-500/80 text-white text-xs px-2 py-1 rounded-full backdrop-blur-sm">

                {wordData.word}

              </div>

            </motion.div>

          ))}

        </div>

      </motion.div>


      {/* Movement instructions */}

      <div className="fixed bottom-6 right-6 bg-black/50 text-white p-4 rounded-xl backdrop-blur-sm z-40">

        <div className="text-xs opacity-75 mb-2">Movement</div>

        <div className="text-sm">

          <div>↑ W - Up</div>

          <div>↓ S - Down</div>

          <div>← A - Left</div>

          <div>→ D - Right</div>

        </div>

      </div>

    </>

  );

}
