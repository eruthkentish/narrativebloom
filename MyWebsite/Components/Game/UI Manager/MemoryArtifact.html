import React, { useState } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { Textarea } from "@/components/ui/textarea";

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

import { Book, Heart, Image, Mail, X, Sparkles } from "lucide-react";


const artifactIcons = {

  letter: Mail,

  photo: Image,

  book: Book,

  memory: Heart

};


export default function MemoryArtifact({ 

  position, 

  type, 

  emotion, 

  prompt, 

  onInteract 

}) {

  const [isActive, setIsActive] = useState(false);

  const [isWriting, setIsWriting] = useState(false);

  const [writtenText, setWrittenText] = useState("");

  const [isCompleted, setIsCompleted] = useState(false);


  const Icon = artifactIcons[type] || Heart;


  const handleActivate = () => {

    setIsActive(true);

    setIsWriting(true);

  };


  const handleSubmit = () => {

    if (writtenText.trim()) {

      setIsCompleted(true);

      setIsWriting(false);

      onInteract({

        type: 'artifact',

        artifactType: type,

        emotion,

        text: writtenText,

        wordCount: writtenText.split(' ').length

      });

      

      // Auto-close after a moment

      setTimeout(() => {

        setIsActive(false);

      }, 2000);

    }

  };


  const handleClose = () => {

    setIsActive(false);

    setIsWriting(false);

  };


  return (

    <>

      {/* Artifact in the world */}

      <motion.div

        className="absolute cursor-pointer z-30"

        style={{

          left: `${position.x}%`,

          top: `${position.y}%`,

          transform: 'translate(-50%, -50%)'

        }}

        animate={{

          y: [0, -10, 0],

          scale: isCompleted ? [1, 1.2, 1] : [1, 1.05, 1]

        }}

        transition={{

          duration: isCompleted ? 0.5 : 3,

          repeat: isCompleted ? 1 : Infinity,

          ease: "easeInOut"

        }}

        onClick={handleActivate}

      >

        <div className={`w-16 h-16 rounded-2xl flex items-center justify-center shadow-xl transition-all duration-500 ${

          isCompleted 

            ? 'bg-gradient-to-br from-emerald-400 to-teal-500' 

            : 'bg-gradient-to-br from-white/80 to-white/60 hover:from-white/90 hover:to-white/70'

        }`}>

          <Icon className={`w-8 h-8 ${isCompleted ? 'text-white' : 'text-gray-700'}`} />

        </div>

        

        {/* Glow effect */}

        <div className={`absolute inset-0 rounded-2xl transition-all duration-500 ${

          isCompleted 

            ? 'shadow-emerald-400/50 shadow-2xl' 

            : 'shadow-white/30 shadow-lg hover:shadow-white/50 hover:shadow-xl'

        }`} />

        

        {/* Interaction hint */}

        {!isCompleted && (

          <motion.div

            initial={{ opacity: 0, y: 10 }}

            animate={{ opacity: 1, y: 0 }}

            className="absolute -bottom-8 left-1/2 transform -translate-x-1/2 bg-black/60 text-white text-xs px-2 py-1 rounded-md whitespace-nowrap"

          >

            Click to explore

          </motion.div>

        )}

      </motion.div>


      {/* Writing Modal */}

      <AnimatePresence>

        {isActive && (

          <motion.div

            initial={{ opacity: 0 }}

            animate={{ opacity: 1 }}

            exit={{ opacity: 0 }}

            className="fixed inset-0 bg-black/70 backdrop-blur-sm flex items-center justify-center p-6 z-50"

          >

            <motion.div

              initial={{ scale: 0.9, opacity: 0 }}

              animate={{ scale: 1, opacity: 1 }}

              exit={{ scale: 0.9, opacity: 0 }}

              className="w-full max-w-2xl"

            >

              <Card className="bg-gradient-to-br from-slate-900/95 to-purple-900/95 border-purple-500/30 text-white">

                <CardHeader className="flex flex-row items-center justify-between">

                  <div className="flex items-center space-x-3">

                    <div className="w-12 h-12 bg-gradient-to-br from-purple-400 to-pink-500 rounded-xl flex items-center justify-center">

                      <Icon className="w-6 h-6 text-white" />

                    </div>

                    <div>

                      <CardTitle className="text-xl capitalize">{type} Memory</CardTitle>

                      <p className="text-sm text-purple-200 capitalize">{emotion} • Processing</p>

                    </div>

                  </div>

                  <Button

                    variant="ghost"

                    size="icon"

                    onClick={handleClose}

                    className="text-purple-300 hover:text-white hover:bg-white/10"

                  >

                    <X className="w-5 h-5" />

                  </Button>

                </CardHeader>

                

                <CardContent className="space-y-6">

                  {isWriting && (

                    <>
