import React from "react";

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

import { TrendingUp, PenTool, Calendar, Heart } from "lucide-react";


export default function WritingStats({ stats }) {

  const statItems = [

    {

      icon: PenTool,

      label: "Entries",

      value: stats.totalEntries,

      color: "text-blue-600",

      bgColor: "bg-blue-100"

    },

    {

      icon: Calendar,

      label: "Day Streak",

      value: stats.streakDays,

      color: "text-emerald-600", 

      bgColor: "bg-emerald-100"

    },

    {

      icon: TrendingUp,

      label: "Words",

      value: stats.wordsWritten?.toLocaleString() || 0,

      color: "text-purple-600",

      bgColor: "bg-purple-100"

    },

    {

      icon: Heart,

      label: "Mood Lift",

      value: stats.moodImprovement > 0 ? `+${stats.moodImprovement}` : stats.moodImprovement,

      color: "text-rose-600",

      bgColor: "bg-rose-100"

    }

  ];


  return (

    <Card className="bg-white/70 backdrop-blur border-gray-200">

      <CardHeader>

        <CardTitle className="text-lg text-gray-800">Your Journey</CardTitle>

      </CardHeader>

      <CardContent className="space-y-4">

        {statItems.map((stat, index) => (

          <div key={index} className="flex items-center justify-between">

            <div className="flex items-center space-x-3">

              <div className={`w-8 h-8 rounded-lg ${stat.bgColor} flex items-center justify-center`}>

                <stat.icon className={`w-4 h-4 ${stat.color}`} />

              </div>

              <span className="text-gray-600 text-sm">{stat.label}</span>

            </div>

            <span className="font-semibold text-gray-800">{stat.value}</span>

          </div>

        ))}

      </CardContent>

    </Card>

  );

}
