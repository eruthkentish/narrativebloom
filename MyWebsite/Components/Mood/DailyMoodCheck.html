import React, { useState } from "react";

import { motion, AnimatePresence } from "framer-motion";

import { Button } from "@/components/ui/button";

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

import { Textarea } from "@/components/ui/textarea";

import { X, Heart, Sun, Zap } from "lucide-react";


const moodEmojis = [

  { value: 1, emoji: "😞", label: "Struggling", color: "text-red-500" },

  { value: 2, emoji: "😐", label: "Low", color: "text-orange-500" },

  { value: 3, emoji: "🙂", label: "Okay", color: "text-yellow-500" },

  { value: 4, emoji: "😊", label: "Good", color: "text-green-500" },

  { value: 5, emoji: "😄", label: "Great", color: "text-blue-500" }

];


const emotionalWords = [

  "peaceful", "anxious", "hopeful", "tired", "excited", "overwhelmed",

  "grateful", "lonely", "confident", "stressed", "creative", "frustrated",

  "calm", "worried", "inspired", "sad", "energetic", "content"

];


export default function DailyMoodCheck({ onSubmit, onClose }) {

  const [mood, setMood] = useState(3);

  const [energy, setEnergy] = useState(3);

  const [stress, setStress] = useState(3);

  const [selectedWords, setSelectedWords] = useState([]);

  const [notes, setNotes] = useState("");


  const toggleWord = (word) => {

    setSelectedWords(prev =>

      prev.includes(word)

        ? prev.filter(w => w !== word)

        : [...prev, word]

    );

  };


  const handleSubmit = () => {

    onSubmit({

      mood_rating: mood,

      energy_level: energy,

      stress_level: stress,

      emotional_words: selectedWords,

      notes

    });

  };


  return (

    <AnimatePresence>

      <motion.div

        initial={{ opacity: 0 }}

        animate={{ opacity: 1 }}

        exit={{ opacity: 0 }}

        className="fixed inset-0 bg-black/50 flex items-center justify-center p-4 z-50"

      >

        <motion.div

          initial={{ scale: 0.9, opacity: 0 }}

          animate={{ scale: 1, opacity: 1 }}

          exit={{ scale: 0.9, opacity: 0 }}

          className="max-w-2xl w-full max-h-[90vh] overflow-y-auto"

        >

          <Card className="bg-white">

            <CardHeader className="flex flex-row items-center justify-between">

              <CardTitle className="flex items-center text-2xl">

                <Heart className="w-6 h-6 text-rose-500 mr-2" />

                Morning Check-in

              </CardTitle>

              <Button

                variant="ghost"

                size="icon"

                onClick={onClose}

                className="text-gray-500 hover:text-gray-700"

              >

                <X className="w-5 h-5" />

              </Button>

            </CardHeader>

            <CardContent className="space-y-8">

              {/* Mood Rating */}

              <div>

                <div className="flex items-center mb-4">

                  <Heart className="w-5 h-5 text-rose-500 mr-2" />

                  <h3 className="text-lg font-semibold">How is your heart feeling?</h3>

                </div>

                <div className="flex justify-between items-center">

                  {moodEmojis.map((item) => (

                    <button

                      key={item.value}

                      onClick={() => setMood(item.value)}

                      className={`flex flex-col items-center p-3 rounded-lg transition-all ${

                        mood === item.value

                          ? 'bg-rose-100 border-2 border-rose-300'

                          : 'hover:bg-gray-100'

                      }`}

                    >

                      <span className="text-2xl mb-1">{item.emoji}</span>

                      <span className={`text-xs font-medium ${item.color}`}>

                        {item.label}

                      </span>

                    </button>

                  ))}

                </div>

              </div>


              {/* Energy Level */}

              <div>

                <div className="flex items-center mb-4">

                  <Zap className="w-5 h-5 text-yellow-500 mr-2" />

                  <h3 className="text-lg font-semibold">What's your energy like?</h3>

                </div>

                <div className="flex space-x-2">

                  {[1, 2, 3, 4, 5].map((level) => (

                    <button

                      key={level}

                      onClick={() => setEnergy(level)}

                      className={`flex-1 p-3 rounded-lg border transition-all ${

                        energy >= level

                          ? 'bg-yellow-100 border-yellow-300'

                          : 'border-gray-200 hover:bg-gray-50'

                      }`}

                    >

                      <Zap className={`w-4 h-4 mx-auto ${

                        energy >= level ? 'text-yellow-600' : 'text-gray-400'

                      }`} />

                    </button>

                  ))}

                </div>

                <div className="flex justify-between text-xs text-gray-500 mt-2">

                  <span>Exhausted</span>

                  <span>Energized</span>

                </div>

              </div>


              {/* Stress Level */}

              <div>

                <div className="flex items-center mb-4">

                  <Sun className="w-5 h-5 text-blue-500 mr-2" />

                  <h3 className="text-lg font-semibold">How stressed do you feel?</h3>

                </div>

                <div className="flex space-x-2">

                  {[1, 2, 3, 4, 5].map((level) => (

                    <button

                      key={level}

                      onClick={() => setStress(level)}

                      className={`flex-1 p-3 rounded-lg border transition-all ${

                        stress >= level
