import React, { useEffect, useState } from "react";

import { Link, useLocation } from "react-router-dom";

import { createPageUrl } from "@/utils";

import { User } from "@/entities/User";

import { 

  BookHeart, 

  Home, 

  PenTool, 

  Heart, 

  Sparkles, 

  TrendingUp,

  Settings,

  Info,

  BookOpen,

  Calendar,

  Mail,

  LogOut,

  Loader2,

  Lock,

  Book, // Added Book icon

  User as UserIcon // Alias User from lucide-react to avoid conflict with entity User

} from "lucide-react";

import { Button } from "@/components/ui/button";


export default function Layout({ children, currentPageName }) {

  const location = useLocation();

  const [user, setUser] = useState(null);

  const [isAuthenticating, setIsAuthenticating] = useState(true);

  const [hasAccess, setHasAccess] = useState(false);


  useEffect(() => {

    const checkUser = async () => {

      try {

        const currentUser = await User.me();

        setUser(currentUser);

        if (currentUser.role === 'admin') {

          setHasAccess(true);

        }

      } catch (error) {

        // User not logged in, redirect to public home page

        window.location.href = createPageUrl('Home');

      } finally {

        setIsAuthenticating(false);

      }

    };

    checkUser();

  }, []);


  const handleLogout = async () => {

    await User.logout();

    window.location.href = createPageUrl('Home');

  };


  const navItems = [

    { name: "Dashboard", path: "Dashboard", icon: Home },

    { name: "Journal", path: "Dashboard?tab=journal", icon: Book },

    { name: "About", path: "About", icon: Info },

    { name: "Writing Studio", path: "WritingStudio", icon: PenTool },

    { name: "Poetry Garden", path: "PoetryGarden", icon: Heart },

    { name: "Reflection", path: "Reflection", icon: Sparkles },

    { name: "Progress", path: "Progress", icon: TrendingUp },

    { name: "Publishing", path: "Publishing", icon: BookOpen },

    { name: "Workshops", path: "Workshops", icon: Calendar },

    { name: "Founder", path: "Founder", icon: UserIcon },

    { name: "Contact", path: "Contact", icon: Mail }

  ];


  if (isAuthenticating) {

    return (

      <div className="min-h-screen bg-gradient-to-br from-rose-50 via-amber-50 to-emerald-50 flex items-center justify-center">

        <Loader2 className="w-12 h-12 animate-spin text-rose-500" />

      </div>

    );

  }


  if (!hasAccess) {

    return (

      <div className="min-h-screen bg-slate-100 flex items-center justify-center">

        <div className="text-center p-8 bg-white rounded-2xl shadow-lg max-w-md mx-auto">

          <Lock className="w-16 h-16 text-red-500 mx-auto mb-6" />

          <h1 className="text-3xl font-bold text-slate-800 mb-4">Access Denied</h1>

          <p className="text-slate-600 mb-8">

            You do not have permission to view this page. This area is restricted to administrators only.

          </p>

          <div className="flex gap-4 justify-center">

            <Link to={createPageUrl("Home")}>

              <Button variant="outline">Go to Homepage</Button>

            </Link>

            <Button onClick={handleLogout}>Logout</Button>

          </div>

        </div>

      </div>

    );

  }


  return (

    <div className="min-h-screen bg-slate-50">

      <style>

        {`

          @import url('https://fonts.googleapis.com/css2?family=Inter:wght@300;400;500;600;700&family=Crimson+Text:ital,wght@0,400;0,600;1,400&display=swap');

          

          :root {

            --primary-blue: #2563eb;

            --primary-slate: #475569; 

            --primary-emerald: #10b981;

            --text-primary: #1f2937;

            --text-secondary: #6b7280;

            --bg-card: rgba(255, 255, 255, 0.8);

            --shadow-soft: 0 8px 32px rgba(0, 0, 0, 0.05);

            --shadow-elevated: 0 20px 60px rgba(0, 0, 0, 0.1);

            --border-subtle: rgba(0, 0, 0, 0.08);

          }

          

          body {

            font-family: 'Inter', -apple-system, BlinkMacSystemFont, sans-serif;

            color: var(--text-primary);

            line-height: 1.7;

            letter-spacing: -0.01em;

          }

          

          .font-serif {

            font-family: 'Crimson Text', Georgia, serif;

          }

          

          .gradient-text {

            background: linear-gradient(135deg, var(--primary-blue), var(--primary-slate));

            -webkit-background-clip: text;

            -webkit-text-fill-color: transparent;

            background-clip: text;

          }

          

          .card-therapy {

            background: var(--bg-card);

            border-radius: 24px;

            border: 1px solid var(--border-subtle);

            backdrop-filter: blur(20px);

            box-shadow: var(--shadow-soft);

          }

          

          .btn-therapy {

            background: var(--primary-blue);

            color: white;

            border: none;

            border-radius: 16px;

            padding: 14px 28px;

            font-weight: 600;

            font-size: 15px;

            transition: all 0.3s cubic-bezier(0.4, 0, 0.2, 1);

            box-shadow: 0 8px 25px rgba(37, 99, 235, 0.2);

          }

          

          .btn-therapy:hover {

            transform: translateY(-2px);

            box-shadow: 0 12px 35px rgba(37, 99, 235, 0.3);

          }

          

          .nav-item {

            transition: all 0.3s cubic-bezier(0.4, 0, 0.2, 1);

            border-radius: 12px;

          }

          

          .nav-item:hover {

            background: rgba(0, 0, 0, 0.04);

            transform: translateY(-1px);

          }

          

          .nav-item.active {

            background: var(--primary-blue);

            color: white;

            box-shadow: 0 8px 25px rgba(37, 99, 235, 0.25);

          }

          

          @media (max-width: 768px) {

            .nav-item {

              border-radius: 8px;

              font-size: 12px;

            }

          }

        `}

      </style>

      

      {/* Navigation */}

      <nav className="bg-white/60 backdrop-blur-xl border-b border-black/5 sticky top-0 z-50">

        <div className="max-w-7xl mx-auto px-6 py-4">

          <div className="flex items-center justify-between">

            <Link to={createPageUrl("Dashboard")} className="flex items-center space-x-3">

              <div className="w-12 h-12 bg-gradient-to-br from-blue-500 to-slate-700 rounded-2xl flex items-center justify-center shadow-lg">

                <BookHeart className="w-6 h-6 text-white" />

              </div>

              <div>

                <h1 className="text-2xl font-bold gradient-text font-serif">Narrative Bloom</h1>

                <p className="text-xs text-gray-500 font-medium">Healing Through Words</p>

              </div>

            </Link>

            

            <div className="hidden xl:flex items-center space-x-1">

              {navItems.map((item) => {

                const isActive = location.pathname.includes(item.path);

                return (

                  <Link

                    key={item.path}

                    to={createPageUrl(item.path)}

                    className={`nav-item flex items-center space-x-2 px-3 py-2 text-sm font-medium transition-all ${

                      isActive

                        ? 'active text-white'

                        : 'text-gray-600 hover:text-gray-800'

                    }`}

                  >

                    <item.icon className="w-4 h-4" />

                    <span>{item.name}</span>

                  </Link>

                );

              })}

            </div>


            <div className="flex items-center space-x-3">

              <button onClick={handleLogout} className="w-10 h-10 bg-gray-100 hover:bg-gray-200 rounded-xl flex items-center justify-center transition-colors" title="Logout">

                <LogOut className="w-4 h-4 text-gray-600" />

              </button>

              <button className="w-10 h-10 bg-gradient-to-br from-blue-500 to-slate-600 rounded-xl flex items-center justify-center shadow-md hover:shadow-lg transition-all" title={user?.email}>

                <UserIcon className="w-4 h-4 text-white" /> {/* Use aliased UserIcon */}

              </button>

            </div>

          </div>

          

          {/* Mobile Navigation */}

          <div className="xl:hidden mt-4">

            <div className="flex space-x-1 overflow-x-auto pb-2">

              {navItems.map((item) => {

                const isActive = location.pathname.includes(item.path);

                return (

                  <Link

                    key={item.path}

                    to={createPageUrl(item.path)}

                    className={`nav-item flex items-center space-x-1 px-2 py-1.5 text-xs font-medium whitespace-nowrap ${

                      isActive

                        ? 'active text-white'

                        : 'text-gray-600'

                    }`}

                  >

                    <item.icon className="w-3 h-3" />

                    <span>{item.name}</span>

                  </Link>

                );

              })}

            </div>

          </div>

        </div>

      </nav>

      

      <main>

        {children}

      </main>

    </div>

  );

}
