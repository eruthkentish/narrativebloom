import { motion, AnimatePresence } from "framer-motion";

import { Card, CardContent } from "@/components/ui/card";

import { Button } from "@/components/ui/button";

import { BookHeart, Play, Pause, Volume2, VolumeX, Sparkles } from "lucide-react";


import Player from "../components/game/Player";

import UIManager from "../components/game/UIManager";

import RealmOfGrief from "../components/game/realms/RealmOfGrief";

import MemoryArtifact from "../components/game/MemoryArtifact";

  { 

    id: 'grief', 

    name: 'The Valley of Letting Go', 

    description: 'Where tears become rivers and loss transforms into acceptance',

    color: 'from-slate-600 to-blue-900',

    symbol: '🌫️'

  },

  { 

    id: 'shame', 

    name: 'The Shadow Gardens', 

    description: 'Where hidden truths bloom in darkness',

    color: 'from-purple-900 to-gray-800',

    symbol: '🌑'

  },

  { 

    id: 'abandonment', 

    name: 'The Echoing Chambers', 

    description: 'Where silence speaks louder than words',

    color: 'from-gray-700 to-slate-900',

    symbol: '🕳️'

  },

  { 

    id: 'forgiveness', 

    name: 'The Healing Springs', 

    description: 'Where bitter waters turn sweet',

    color: 'from-emerald-400 to-teal-600',

    symbol: '🌿'

  },

  { 

    id: 'hope', 

    name: 'The Dawn Meadows', 

    description: 'Where new light breaks the horizon',

    color: 'from-yellow-300 to-orange-500',

    symbol: '🌅'

  }

];


export default function ProcessingGame() {

  const [gameState, setGameState] = useState('menu'); // menu, playing, paused, realm-select

  const [currentRealm, setCurrentRealm] = useState(null);

  const [playerProgress, setPlayerProgress] = useState({

    realmsUnlocked: ['grief'], // Start with grief realm

    artifacts: [],

    writtenWords: 0,

    emotionalJourney: []

  });

  const [soundEnabled, setSoundEnabled] = useState(true);

  const [isLoading, setIsLoading] = useState(false);


  const startGame = () => {

    setGameState('realm-select');

  };


  const enterRealm = (realmId) => {

    setIsLoading(true);

    setCurrentRealm(realmId);

    setTimeout(() => {

      setIsLoading(false);

      setGameState('playing');

    }, 2000);

  };


  const exitToMenu = () => {

    setCurrentRealm(null);

    setGameState('menu');

  };


  const renderMainMenu = () => (

    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-indigo-900 flex items-center justify-center p-6">

      <div className="absolute inset-0 overflow-hidden">

        <div className="absolute inset-0 bg-gradient-to-br from-transparent via-purple-900/20 to-transparent animate-pulse" />

        <div className="absolute top-0 left-0 w-full h-full bg-[url('data:image/svg+xml,%3Csvg width="60" height="60" viewBox="0 0 60 60" xmlns="http://www.w3.org/2000/svg"%3E%3Cg fill="none" fill-rule="evenodd"%3E%3Cg fill="%23ffffff" fill-opacity="0.05"%3E%3Ccircle cx="30" cy="30" r="2"/%3E%3C/g%3E%3C/g%3E%3C/svg%3E')] opacity-30" />

      </div>


      <motion.div

        initial={{ opacity: 0, y: 30 }}

        animate={{ opacity: 1, y: 0 }}

        transition={{ duration: 1 }}

        className="text-center z-10"

      >

        <div className="mb-8">

          <motion.div

            animate={{ 

              rotate: [0, 5, -5, 0],

              scale: [1, 1.05, 1]

            }}

            transition={{ 

              duration: 4,

              repeat: Infinity,

              repeatType: "reverse"

            }}

            className="w-24 h-24 bg-gradient-to-br from-purple-400 to-pink-500 rounded-3xl flex items-center justify-center mx-auto mb-6 shadow-2xl"

          >

            <BookHeart className="w-12 h-12 text-white" />

          </motion.div>

        </div>


        <motion.h1

          initial={{ opacity: 0 }}

          animate={{ opacity: 1 }}

          transition={{ delay: 0.5 }}

          className="text-6xl md:text-7xl font-serif font-bold text-white mb-6 leading-tight"

        >

          Processing

        </motion.h1>

        

        <motion.p

          initial={{ opacity: 0 }}

          animate={{ opacity: 1 }}

          transition={{ delay: 0.7 }}

          className="text-xl text-purple-200 mb-4 max-w-2xl mx-auto italic"

        >

          A narrative exploration of the emotional self

        </motion.p>

        

        <motion.p

          initial={{ opacity: 0 }}

          animate={{ opacity: 1 }}

          transition={{ delay: 0.9 }}

          className="text-lg text-purple-300/80 mb-12 max-w-3xl mx-auto leading-relaxed"

        >

          Journey through dreamlike realms of emotion. Write your story. Transform pain into power. 

          This is not about winning—it's about feeling, expressing, and healing.

        </motion.p>


        <motion.div

          initial={{ opacity: 0, y: 20 }}

          animate={{ opacity: 1, y: 0 }}

          transition={{ delay: 1.1 }}

          className="space-y-4"

        >

          <Button

            onClick={startGame}

            className="bg-gradient-to-r from-purple-500 to-pink-500 hover:from-purple-600 hover:to-pink-600 text-white px-12 py-4 rounded-2xl text-lg font-medium shadow-2xl transform hover:scale-105 transition-all duration-300"

          >

            <Play className="mr-3 w-5 h-5" />

            Begin Journey

          </Button>

          

          <div className="flex justify-center space-x-4 mt-8">

            <button

              onClick={() => setSoundEnabled(!soundEnabled)}

              className="p-3 rounded-full bg-white/10 text-white hover:bg-white/20 transition-colors"

            >

              {soundEnabled ? <Volume2 className="w-5 h-5" /> : <VolumeX className="w-5 h-5" />}

            </button>

          </div>

        </motion.div>

      </motion.div>

    </div>

  );


  const renderRealmSelect = () => (

    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-indigo-900 py-12">

      <div className="max-w-6xl mx-auto px-6">

        <motion.div

          initial={{ opacity: 0, y: -30 }}

          animate={{ opacity: 1, y: 0 }}

          className="text-center mb-16"

        >

          <h2 className="text-4xl font-serif font-bold text-white mb-4">

            Choose Your Emotional Realm

          </h2>

          <p className="text-purple-200 text-lg">

            Each realm represents a different landscape of the heart

          </p>

        </motion.div>


        <div className="grid md:grid-cols-2 lg:grid-cols-3 gap-8">

          {emotionalRealms.map((realm, index) => (

            <motion.div

              key={realm.id}

              initial={{ opacity: 0, y: 50 }}

              animate={{ opacity: 1, y: 0 }}

              transition={{ delay: index * 0.1 }}

              className={`relative overflow-hidden rounded-3xl ${

                playerProgress.realmsUnlocked.includes(realm.id) 

                  ? 'cursor-pointer' 

                  : 'cursor-not-allowed opacity-50'

              }`}

              onClick={() => playerProgress.realmsUnlocked.includes(realm.id) && enterRealm(realm.id)}

            >

              <Card className={`bg-gradient-to-br ${realm.color} border-none text-white hover:shadow-2xl transition-all duration-500 transform hover:scale-105 group`}>

                <CardContent className="p-8 text-center relative">

                  <div className="absolute inset-0 bg-black/20 group-hover:bg-black/10 transition-all duration-500" />

                  <div className="relative z-10">

                    <div className="text-4xl mb-4">{realm.symbol}</div>

                    <h3 className="text-xl font-bold mb-3">{realm.name}</h3>

                    <p className="text-white/80 text-sm leading-relaxed mb-4">

                      {realm.description}

                    </p>

                    {!playerProgress.realmsUnlocked.includes(realm.id) && (

                      <div className="absolute inset-0 bg-black/60 flex items-center justify-center rounded-3xl">

                        <p className="text-white/80 text-sm">Realm Locked</p>

                      </div>

                    )}

                  </div>

                </CardContent>

              </Card>

            </motion.div>

          ))}

        </div>


        <div className="text-center mt-12">

          <Button

            onClick={exitToMenu}

            variant="outline"

            className="border-purple-400 text-purple-300 hover:bg-purple-400/10"

          >

            Return to Menu

          </Button>

        </div>

      </div>

    </div>

  );


  const renderLoading = () => (

    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-indigo-900 flex items-center justify-center">

      <motion.div

        initial={{ opacity: 0 }}

        animate={{ opacity: 1 }}

        className="text-center"

      >

        <motion.div

          animate={{ rotate: 360 }}

          transition={{ duration: 2, repeat: Infinity, ease: "linear" }}

          className="w-16 h-16 border-4 border-purple-400 border-t-transparent rounded-full mx-auto mb-8"

        />

        <h3 className="text-2xl font-serif text-white mb-4">

          Entering the {emotionalRealms.find(r => r.id === currentRealm)?.name}...

        </h3>

        <p className="text-purple-300">

          Preparing your emotional landscape

        </p>

      </motion.div>

    </div>

  );


  const renderGame = () => (

    <div className="relative min-h-screen overflow-hidden">

      {currentRealm === 'grief' && <RealmOfGrief />}

      

      <UIManager 

        playerProgress={playerProgress}

        currentRealm={currentRealm}

        onExit={exitToMenu}

        soundEnabled={soundEnabled}

        setSoundEnabled={setSoundEnabled}

      <Player 

        realm={currentRealm}

        progress={playerProgress}

        setProgress={setPlayerProgress}

      />

    </div>

  );


  return (

    <AnimatePresence mode="wait">

      {gameState === 'menu' && (

        <motion.div key="menu" exit={{ opacity: 0 }}>

          {renderMainMenu()}

        </motion.div>

      )}

      

      {gameState === 'realm-select' && (

        <motion.div key="realm-select" exit={{ opacity: 0 }}>

          {renderRealmSelect()}

        </motion.div>

      )}

      

      {isLoading && (

        <motion.div key="loading" exit={{ opacity: 0 }}>

          {renderLoading()}

        </motion.div>

      )}

      

      {gameState === 'playing' && !isLoading && (

        <motion.div key="playing" initial={{ opacity: 0 }} animate={{ opacity: 1 }}>

          {renderGame()}

        </motion.div>

      )}

    </AnimatePresence>

  );

}
